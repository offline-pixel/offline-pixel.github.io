Complete Pine Script Code


//@version=5
strategy("Breakout Strategy - Russell 2000", overlay=true,
         initial_capital=25000, default_qty_type=strategy.percent_of_equity,
         default_qty_value=10, commission_type=strategy.commission.percent,
         commission_value=0.05)

// Indicators
basis = ta.sma(close, 20)
dev = 2 * ta.stdev(close, 20)
upper = basis + dev
lower = basis - dev
volumeAvg = ta.sma(volume, 20)
atr = ta.atr(14)

// Entry Conditions
long_condition = close > upper and volume > 1.5 * volumeAvg
short_condition = close < lower and volume > 1.5 * volumeAvg

stop_mult = 2
long_stop = close - (atr * stop_mult)
short_stop = close + (atr * stop_mult)

// Strategy Execution
if long_condition
    strategy.entry("Breakout Long", strategy.long)
    strategy.exit("Exit Long", "Breakout Long", stop=long_stop)

if short_condition
    strategy.entry("Breakout Short", strategy.short)
    strategy.exit("Exit Short", "Breakout Short", stop=short_stop)
