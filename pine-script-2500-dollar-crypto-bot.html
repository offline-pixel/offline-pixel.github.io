Complete Pine Script Code


//@version=5
strategy("$2500/Day Crypto Bot", overlay=true, 
         initial_capital=10000, default_qty_type=strategy.percent_of_equity, 
         default_qty_value=10, commission_type=strategy.commission.percent, 
         commission_value=0.1)

// Chande Momentum Oscillator (CMO)
cmo_length = 14
cmo = ta.cmo(close, cmo_length)

// Normalized ATR
atr_length = 14
atr = ta.atr(atr_length)
natr = atr / close
valid_volatility = natr > 0.01

// Entry Conditions
long_condition = ta.crossover(cmo, -50) and valid_volatility
short_condition = ta.crossunder(cmo, 50) and valid_volatility

// Adaptive Stop-Loss
stop_mult = 1.5
long_stop = low - (atr * stop_mult)
short_stop = high + (atr * stop_mult)

// Strategy Execution
if long_condition
    strategy.entry("Long", strategy.long)
    strategy.exit("Exit Long", "Long", stop=long_stop)

if short_condition
    strategy.entry("Short", strategy.short)
    strategy.exit("Exit Short", "Short", stop=short_stop)
