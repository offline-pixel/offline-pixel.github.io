Complete Pine Script Code


//@version=5
strategy("Breakout Strategy", overlay=true, 
         initial_capital=40000, default_qty_type=strategy.percent_of_equity, 
         default_qty_value=10, commission_type=strategy.commission.percent, 
         commission_value=0.1)

// Parameters
length = input.int(20, title="Channel Length")
atr_length = input.int(14, title="ATR Length")
atr_mult = input.float(1.5, title="ATR Multiplier")

// Indicators
upper_band = ta.highest(high, length)
lower_band = ta.lowest(low, length)
atr = ta.atr(atr_length)

// Entry Conditions
long_condition = close > upper_band[1]
short_condition = close < lower_band[1]

// Strategy Execution
if long_condition
    strategy.entry("Long", strategy.long)
    strategy.exit("Exit Long", "Long", stop=close - atr * atr_mult)

if short_condition
    strategy.entry("Short", strategy.short)
    strategy.exit("Exit Short", "Short", stop=close + atr * atr_mult)
