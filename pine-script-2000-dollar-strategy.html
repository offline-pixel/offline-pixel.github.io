Complete Pine Script Code


//@version=5

strategy("$2000/Day Institutional Algo", overlay=true, 
         initial_capital=25000, default_qty_type=strategy.percent_of_equity, 
         default_qty_value=2, commission_type=strategy.commission.percent, 
         commission_value=0.04)


// ======== SMART MONEY DETECTION ========

cumDelta = ta.cum(close > open ? volume : -volume)

vProfile = ta.valuewhen(ta.change(timeframe.period("D")), volume, 0)

smartMoney = cumDelta > vProfile * 1.25


// ======== MULTI-TIMEFRAME CONFIRMATION ========

htfTrend = request.security(syminfo.tickerid, "60", 
           ta.ema(close, 50) > ta.ema(close, 200) ? 1 : 
           ta.ema(close, 50) < ta.ema(close, 200) ? -1 : 0)


// ======== ENTRY LOGIC ========

longCondition = smartMoney and htfTrend == 1 and 
                ta.crossover(ta.ema(close, 9), ta.ema(close, 21)) and 
                volume > ta.sma(volume, 20)[1] * 1.8

shortCondition = smartMoney and htfTrend == -1 and 
                 ta.crossunder(ta.ema(close, 9), ta.ema(close, 21)) and 
                 volume > ta.sma(volume, 20)[1] * 1.8


// ======== RISK MANAGEMENT ========

strategy.entry("Long", strategy.long, when=longCondition)

strategy.entry("Short", strategy.short, when=shortCondition)

strategy.exit("Exit", "Long", loss=1.5, profit=5.7)

strategy.exit("Exit", "Short", loss=1.5, profit=5.7)
