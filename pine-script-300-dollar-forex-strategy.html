Complete Pine Script Code


//@version=5
strategy("$300/Day Forex Scalper", overlay=true, 
    initial_capital=3000, 
    default_qty_type=strategy.percent_of_equity, 
    default_qty_value=3)

// EMA crossover logic
ema5 = ta.ema(close, 5)
ema20 = ta.ema(close, 20)

rsi = ta.rsi(close, 14)

longCondition = ta.crossover(ema5, ema20) and rsi > 55
shortCondition = ta.crossunder(ema5, ema20) and rsi < 45

strategy.entry("Long", strategy.long, when=longCondition)
strategy.entry("Short", strategy.short, when=shortCondition)

strategy.exit("ExitLong", "Long", loss=0.8, profit=2.0)
strategy.exit("ExitShort", "Short", loss=0.8, profit=2.0)
