  2. Volume spike filter to avoid dead sessions
  3. ATR-based tight stop-loss for minimal exposure

Complete Pine Script Code


//@version=5
strategy("$1000/Week Scalping Bot", overlay=true,
         initial_capital=12000, default_qty_type=strategy.percent_of_equity,
         default_qty_value=15, commission_type=strategy.commission.percent,
         commission_value=0.1)

// Indicators
emaFast = ta.ema(close, 5)
emaSlow = ta.ema(close, 13)
vol = volume
volMA = ta.sma(vol, 20)
atr = ta.atr(14)

// Entry Conditions
longCond = ta.crossover(emaFast, emaSlow) and vol > volMA
shortCond = ta.crossunder(emaFast, emaSlow) and vol > volMA

// Tight Risk Management
stopMult = 1.2
longStop = close - atr * stopMult
shortStop = close + atr * stopMult

// Execution
if longCond
    strategy.entry("Long", strategy.long)
    strategy.exit("Exit Long", "Long", stop=longStop)

if shortCond
    strategy.entry("Short", strategy.short)
    strategy.exit("Exit Short", "Short", stop=shortStop)
